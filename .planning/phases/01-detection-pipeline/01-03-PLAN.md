    1. Create src/egg_counter/detector.py with class EggDetector:
       - __init__(self, model_path: str, tracker_config: str = "config/bytetrack_eggs.yaml", confidence: float = 0.5):
         - from ultralytics import YOLO
         - self.model = YOLO(model_path)
         - self.tracker_config = tracker_config
         - self.confidence = confidence
         - self.frame_count = 0

       - detect_and_track(self, frame) -> dict:
         - self.frame_count += 1
         - Runs self.model.track(frame, persist=True, tracker=self.tracker_config, conf=self.confidence, verbose=False)
         - Extracts from results[0]:
           - track_ids: list[int] from results[0].boxes.id.int().cpu().tolist() if results[0].boxes.id is not None else []
           - boxes: list of [x1,y1,x2,y2] from results[0].boxes.xyxy.cpu().numpy().tolist()
           - confidences: list[float] from results[0].boxes.conf.cpu().tolist()
           - classes: list[int] from results[0].boxes.cls.int().cpu().tolist() (for multi-class model support)
         - Returns {"track_ids": track_ids, "boxes": boxes, "confidences": confidences, "classes": classes, "frame_number": self.frame_count}
         - If no detections (boxes.id is None): returns {"track_ids": [], "boxes": [], "confidences": [], "classes": [], "frame_number": self.frame_count}

       - detect_once(self, frame) -> dict:
         - For restart initialization (D-08): runs model.predict (not track) on a single frame
         - Returns same format as detect_and_track
         - Does NOT use persist=True (no tracking state initialization)

       - reset_tracker(self):
         - Resets tracking state by re-loading the model's tracker
         - self.model.predictor = None (forces re-init on next track call)

    1. Create src/egg_counter/pipeline.py with class EggCounterPipeline:

       __init__(self, settings: dict, zone_config: dict):
         - self.settings = settings
         - self.zone_config = zone_config
         - self.detector: EggDetector = None  # Initialized in setup()
         - self.tracker = EggTracker(stability_seconds=settings.get("stability_seconds", 3))
         - self.classifier = SizeClassifier(zone_config)
         - self.logger = EggEventLogger(settings.get("log_dir", "logs"))
         - self.running = False
         - self.frame_rate = settings.get("frame_rate", 3)

       setup(self, model_path: str):
         - Initializes self.detector = EggDetector(model_path, tracker_config=self.settings.get("bytetrack_config", "config/bytetrack_eggs.yaml"), confidence=self.settings.get("confidence_threshold", 0.5))

       _initialize_existing_eggs(self, frame):
         - Per D-08: Run detect_once on first frame
         - Extract track_ids from result
         - Filter to only eggs in zone (using is_in_zone)
         - Call self.tracker.initialize_from_existing(in_zone_track_ids)
         - If any found, print f"Startup: {len(in_zone_track_ids)} existing eggs detected, marked as already counted"

       process_frame(self, frame, timestamp: float) -> list[dict]:
         - Core single-frame processing (extracted for testability)
         - Calls self.detector.detect_and_track(frame)
         - For each detection, compute in_zone = is_in_zone(box, self.zone_config)
         - Calls self.tracker.process_detections(track_ids, boxes, in_zone_flags, timestamp)
         - For each tracker result with action=="count":
           - size, raw_mm = self.classifier.classify(result["bbox"])
           - Find matching confidence from detector results
           - Calls self.logger.log_egg_detected(track_id=result["track_id"], size=size, confidence=confidence, bbox=result["bbox"], size_method="bbox_ratio", raw_measurement_mm=raw_mm, frame_number=detector_result["frame_number"])
         - For each tracker result with action=="collected":
           - Calls self.logger.log_eggs_collected(count=result["count"])
         - Returns list of logged events

       run(self, model_path: str, camera_index: int = 0):
         - self.setup(model_path)
         - Opens camera: cap = cv2.VideoCapture(camera_index)
           - On Linux (sys.platform == "linux"): use cv2.VideoCapture(camera_index, cv2.CAP_V4L2)
         - cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
         - cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)
         - Print "Egg Counter started. Press Ctrl+C to stop."
         - Read first frame, call _initialize_existing_eggs(frame)
         - self.running = True
         - Main loop (while self.running and cap.isOpened()):
           a. Check daylight if location configured: if not is_daylight(lat, lon): wait_for_daylight(lat, lon); continue
           b. Read frame. If not ret: print warning, sleep 1s, retry
           c. Call process_frame(frame, time.time())
           d. Sleep to maintain target frame_rate: time.sleep(1.0 / self.frame_rate)
         - Finally: cap.release(), print "Egg Counter stopped."

       stop(self):
         - self.running = False

         - parser = argparse.ArgumentParser(description="Egg Counter - YOLO-based egg detection and counting")
